#include <stdio.h>
#include <stdlib.h>
#include <sys/types.h>
#include <unistd.h>
#include <string.h>
#include <sys/wait.h>
#include <sys/stat.h>
#include <fcntl.h>

int main() {
    printf("parent: spawning...\n\n");

    pid_t child1_pid = fork();
    if (child1_pid == 0) {
        char* cmd1 = "./program1";
        char* argv1[] = {"program1", NULL};
        execvp(cmd1, argv1);
        perror("error exe program1");
        exit(1);
    }

    pid_t child2_pid = fork();
    if (child2_pid == 0) {
        char* cmd2 = "./program2";
        char* argv2[] = {"program2", NULL};
        execvp(cmd2, argv2);
        perror("error exe program2");
        exit(1);
    }

    waitpid(child1_pid, NULL, 0);
    waitpid(child2_pid, NULL, 0);
    printf("\nparent: both children done. exiting.\n");
    return 0;
}

4)
dup2() works by telling the OS to take the existing file descriptor(newfd) and completely overwrite it with a cop of this other one(oldfd). So in program1 I'm going to use dup2(fd[1], stdout_fileno); which will close the connection to the screen and replace it with the write end of the pipe. as for program2 ill use dup2(fd[0], stdin_fileno) which will close the connection to the keyboard and replace it with the read end of the pipe.
